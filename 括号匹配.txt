
#include<stdio.h>
#include<string.h>
#include<iostream>
#include<stack>
using namespace std;

int cop(char x)
{
    if(x == '(' || x == ')')
    {
        return 1;
    }

    else if(x == '{' || x == '}')
    {
        return 2;
    }

    else if(x == '[' || x == ']')
    {
        return 3;
    }
}

int main(void)
{
    int flag, n, i;
    stack<char>p;
    char str[100];
    while(gets(str) != NULL)
    {
        n = strlen(str);
        flag = 0;
        for(i = 0; i < n; i++)
        {
            //printf("%c\n", str[i]);
            if(p.empty() && (str[i] == ')' || str[i] == '}' || str[i] == ']'))
            {
                flag = 1;
                break;
            }

            else if(str[i] == '(' || str[i] == '{' || str[i] == '[')
            {
                p.push(str[i]);
            }

            else if(str[i] == ')' || str[i] == '}' || str[i] == ']')
            {

                if(cop(str[i]) == cop(p.top()))
                {
                    p.pop();
                }

                else
                {
                    flag = 1;
                    break;
                }
            }
        }


        if(!p.empty() || flag)
        {
            printf("no\n");
        }

        else
        {
            printf("no\n");
        }

    }

    return 0;
}
